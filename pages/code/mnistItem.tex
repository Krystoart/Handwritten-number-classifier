\begin{minted}{csharp}
    public MnistItem(int width, int height, byte[][] pixels, byte label)
    {
        this.Width = width;
        this.Height = height;
        this.Label = label;

        this.Pixels = new float[height*width];

        int counter = 0;
        for (int i = 0; i < height; ++i)
        {
            for (int j = 0; j < width; ++j)
            {
                this.Pixels[counter] = ((float)pixels[i][j]) / byte.MaxValue;
                counter += 1;
            }
        }
    }
\end{minted}
